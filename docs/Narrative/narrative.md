<script>
// JavaScript para el carrusel de tarjetas
document.addEventListener('DOMContentLoaded', function() {
    const slides = document.querySelectorAll('.card-slide');
    const dots = document.querySelectorAll('.dot');
    const prevBtn = document.querySelector('.prev-btn');
    const nextBtn = document.querySelector('.next-btn');
    let currentSlide = 0;
    const totalSlides = slides.length;
    
    // Función para mostrar un slide específico
    function showSlide(index) {
        // Remover clase active de todos los slides
        slides.forEach(slide => {
            slide.classList.remove('active');
        });
        
        // Remover clase active de todos los dots
        dots.forEach(dot => {
            dot.classList.remove('active');
        });
        
        // Asegurar que el índice esté dentro de los límites
        if (index >= totalSlides) {
            currentSlide = 0;
        } else if (index < 0) {
            currentSlide = totalSlides - 1;
        } else {
            currentSlide = index;
        }
        
        // Agregar clase active al slide actual
        slides[currentSlide].classList.add('active');
        dots[currentSlide].classList.add('active');
    }
    
    // Evento para botón siguiente
    nextBtn.addEventListener('click', function() {
        showSlide(currentSlide + 1);
    });
    
    // Evento para botón anterior
    prevBtn.addEventListener('click', function() {
        showSlide(currentSlide - 1);
    });
    
    // Evento para los dots
    dots.forEach(dot => {
        dot.addEventListener('click', function() {
            const slideIndex = parseInt(this.getAttribute('data-slide'));
            showSlide(slideIndex);
        });
    });
    
    // Auto-avance cada 8 segundos
    let slideInterval = setInterval(function() {
        showSlide(currentSlide + 1);
    }, 8000);
    
    // Pausar auto-avance al interactuar
    const sliderContainer = document.querySelector('.cards-slider-container');
    sliderContainer.addEventListener('mouseenter', function() {
        clearInterval(slideInterval);
    });
    
    sliderContainer.addEventListener('mouseleave', function() {
        slideInterval = setInterval(function() {
            showSlide(currentSlide + 1);
        }, 8000);
    });
    
    // Inicializar el carrusel
    showSlide(0);
});
</script>
